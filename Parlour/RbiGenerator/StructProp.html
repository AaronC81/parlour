def to_prop_call
  call = "prop :#{name}, #{String === @type ? @type : @type.generate_rbi}"

  EXTRA_PROPERTIES.each do |extra_property|
    value = send extra_property
    call += ", #{extra_property}: #{value}" unless value.nil?
  end

  call
end
